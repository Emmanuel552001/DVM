<?php

#[Route("/equipements/conducteur/{id}", name: "equipements_par_conducteur", methods: ["GET"])]
    public function getPrixTotalEquipements(int $id, EquipementVehiculeRepository $equipementVehiculeRepository): JsonResponse
    {
        $equipements = $equipementVehiculeRepository->findByConducteur($id);
        $prixTotal = array_reduce($equipements, function ($total, $equipement) {
            return $total + ($equipement->getEqveQuantite() * $equipement->getEqveEquipement()->getEqPrix());
        }, 0);
    
        return $this->json(['prix_total' => $prixTotal]);
    }


    #[Route('/equipements/vehicule/{id}', name: 'equipements_par_vehicule', methods: ['GET'])]
public function getEquipementsParVehicule(int $id, EquipementVehiculeRepository $equipementVehiculeRepository): JsonResponse
{
    $equipements = $equipementVehiculeRepository->findByVehicule($id);
    
    $equipementList = array_map(function ($equipementVehicule) {
        return [
            'id' => $equipementVehicule->getEqVeId(),
            'libelle' => $equipementVehicule->getEqVeEquipement()->getEqLibelle(),
            'quantite' => $equipementVehicule->getEqVeQuantite(),
            'prix' => $equipementVehicule->getEqVeEquipement()->getEqPrix()
        ];
    }, $equipements);

    return $this->json(['equipements' => $equipementList]);
}
